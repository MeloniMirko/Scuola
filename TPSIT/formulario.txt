// JAVASCRIPT

// Richiesta dati all'utente tramite finestra del browser
let nome = prompt("Come ti chiami?");
let numero = prompt("Dammi un numero")*1;

// Visualizzazione dati su finestra dl browser
alert("Ciao " +nome);
alert("Benvenuto nel mio sito");

// Recupero dati inseriti dall'utente nei campi input dell'html
let testoUtente = document.getElementById("testo").value;
let numero = document.getElementById("num").value;

// Visualizzazione dinamica di dati nella pagina html
document.getElementById("dati").innerHTML = testoUtente + " "+numero;

// Visualizzazione dinamica di dati nella pagina html senza sovrascrivere il contenuto
document.getElementById("dati").innerHTML = document.getElementById("dati").innerHTML +"<br>"+ testoUtente + " "+numero;
